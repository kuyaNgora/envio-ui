### Usage
#### With Initial Name
<Source code={`
import React from "react";
import { Avatar, AvatarProps } from "@envio-ui/design";

export const Example: React.FC<AvatarProps> = () => {
  return (
    <Avatar
      variant="primary"
      name="Lorem Ipsum"
      size={50}
    />
  );
};
`} dark language="tsx" />

#### With Initial Image
<Source code={`
import React from "react";
import { Avatar, AvatarProps } from "@envio-ui/design";

export const Example: React.FC<AvatarProps> = () => {
  return (
    <Avatar
      src="https://s3-ap-southeast-1.amazonaws.com/envio.co.id/production-files/1092hbc0c059h6c.png?v=103024224441203141333411220"
      name="Lorem Ipsum"
      size={50}
    />
  );
};
`} dark language="tsx" />

#### With Icons
<Source code={`
import React from "react";
import { Avatar, AvatarProps } from "@envio-ui/design";
import { IconPerson } from "@envio-ui/icons";

export const Example: React.FC<AvatarProps> = () => {
  return (
    <Avatar
      variant="primary"
      name="Lorem Ipsum"
      size={50}
      icon={IconPerson}
      color="#fff"
    />
  );
};
`} dark language="tsx" />

#### With Initial Presence
<Source code={`
import React from "react";
import { Box, Avatar, AvatarProps } from "@envio-ui/design";
import { IconPerson } from "@envio-ui/icons";

export const Example: React.FC<AvatarProps> = () => {
  return (
    <Avatar
      variant="primary"
      name="Lorem Ipsum"
      size={50}
      icon={IconPerson}
      presence={<Box width="100%" height="100%" backgroundColor={"red07"} />}
      color="#fff"
    />
  );
};
`} dark language="tsx" />
