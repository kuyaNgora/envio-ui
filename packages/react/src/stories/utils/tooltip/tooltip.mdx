# Tooltip
A popup that displays information related to an element when the element receives keyboard focus or the mouse hovers over it.

<Canvas>
  <Story of={TooltipStories.Example} />
</Canvas>

### Usage
<Source code={`
import React from "react";
import { Tooltip, TooltipProps } from "@envio-ui/design";

export const Example: React.FC<TooltipProps> = () => {
  return (
    <Tooltip
      placement="top"
      trigger={<div>trigger</div>}
      content={
        <div>
          Lorem ipsum dolor sit amet, consectetur adipiscing elit.
        </div>
      }
    />
  );
};
`} of={TooltipStories.Example} dark language="tsx" />
