# Alert Dialog
A modal dialog that interrupts the user with important content and expects a response.

<Canvas>
  <Story of={AlertDialogStories.Example} />
</Canvas>


### Usage
<Source code={`
import React from "react";
import { Button, AlertDialog } from "@envio-ui/design";

export const Example: React.FC = () => {
  const [show, setShow] = React.useState(false);

  return (
    <AlertDialog open={show}>
      <AlertDialog.Trigger asChild>
        <Button onClick={() => setShow(true)}>action</Button>
      </AlertDialog.Trigger>
      <AlertDialog.Portal>
        <AlertDialog.Overlay />
        <AlertDialog.Content
          cancelText="Cancel"
          actionText="Yes, delete account"
          onCancel={() => setShow(false)}
          onSubmit={() => console.log("aksi")}
        >
          <AlertDialog.Title>Are you absolutely sure?</AlertDialog.Title>
          <AlertDialog.Description>
            This action cannot be undone. This will permanently delete your
            account and remove your data from our servers.
          </AlertDialog.Description>
        </AlertDialog.Content>
      </AlertDialog.Portal>
    </AlertDialog>
  );
};
`} of={AlertDialogStories.Example} dark language="tsx" />
