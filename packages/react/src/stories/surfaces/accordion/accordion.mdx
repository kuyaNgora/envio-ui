# Accordion
A vertically stacked set of interactive headings that each reveal an associated section of content.

### Anatomy
<Source
    code={
`<Accordion>
    <Accordion.Item>
        <Accordion.Header />
        <Accordion.Content />
    </Accordion.Item>
</Accordion>`
    }
    dark
    language="tsx"
/>

import { Accordion } from "@envio-ui/design";

export const SingleAccordion: StoryFn<{}> = () => {
