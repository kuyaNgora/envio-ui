# Usage

<Source code={
`
import React from 'react';
import App from "./App";
import {EnvioProvider} from '@react-ui/react';

const root = ReactDOM.createRoot(document.getElementById("root"));
root.render(
  <React.StrictMode>
    <EnvioProvider>
      <App />
    </EnvioProvider>
  </React.StrictMode>
);
`
